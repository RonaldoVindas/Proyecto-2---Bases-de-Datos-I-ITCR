/*========== QUERY: LESS EXPENSIVE PRODUCTS BY TYPE=============*/
/*Author: Ronaldo Vindas B.  2018238697*/

USE `pr`;
DROP procedure IF EXISTS `get_less_expensive_products_by_type`;

DELIMITER $$
USE `pr`$$
CREATE PROCEDURE `get_less_expensive_products_by_type` (IN pQuantity INT)
BEGIN
	SELECT product_type.type, MIN(price), product.name
	FROM product
	INNER JOIN product_type
	ON product.id_product_type = product_type.id_product_type 
	GROUP BY product_type.type 
	ORDER BY price DESC
    LIMIT pQuantity;
END$$

DELIMITER ;

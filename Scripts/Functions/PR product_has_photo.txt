DROP function IF EXISTS `get_id_product_php`;

DELIMITER $$
USE `pe`$$
CREATE FUNCTION `get_id_product_php` (pId INT)
RETURNS INTEGER
DETERMINISTIC
BEGIN


	DECLARE EXIT HANDLER FOR 1231 SELECT 'Variable cannot be set to that value.' Message;
	DECLARE EXIT HANDLER FOR 1232 SELECT 'Incorrect argument type to variable.' Message;
	DECLARE EXIT HANDLER FOR 1105 SELECT 'Unknown error encountered.' Message;



	DECLARE vcValue INT;
    SET vcValue = (SELECT id_product FROM product_has_photo WHERE id_photo = pId);
	RETURN vcValue;
END$$

DELIMITER ;

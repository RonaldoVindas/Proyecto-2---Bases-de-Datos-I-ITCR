USE `pe`;

/*=============Getters===========================*/


DROP function IF EXISTS `get_id_product`;

DELIMITER $$
USE `pe`$$
CREATE FUNCTION `get_id_product` (pId INT)
RETURNS INTEGER
DETERMINISTIC
BEGIN
	DECLARE vcValue INT;
    SET vcValue = (SELECT id_product FROM shopping_log_has_product WHERE id_shopping_log = pId);
	RETURN vcValue;
END$$

DELIMITER ;

USE `pe`;

/*=============Getters===========================*/

DROP function IF EXISTS `get_person_first_name`;

DELIMITER $$
USE `pe`$$
CREATE FUNCTION `get_person_first_name` (pId INT)
RETURNS VARCHAR(45)
DETERMINISTIC
BEGIN
	DECLARE vcName VARCHAR(45);
    SET vcName = (SELECT first_name FROM person WHERE identification = pId);
	RETURN vcName;
END$$

DELIMITER ;
